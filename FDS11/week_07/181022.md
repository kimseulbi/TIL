function limit (min, max, input) {
  if (min> input){
    return min;
  } else if (max < input){
    return max;

function isPrime(x) {
  // 소수: 1과 자기자신밖에 약수가 없는 수
  // -> 1과 자기자신이 아닌 약수가 하나라도 있으면 소수가 아니다.

  }

}

  const n = i +1;
  const line = ' '.repeat(height - n) + '* '.repeat(n);
  console.log(line)
}

function print(height) {
  for (let i = 0; i<height; i++){
    printLine(height, i)
  }
